  export const ease = {
    snap: [0.19, 1, 0.22, 1],       // fast out, used for entrances
    out: [0.0, 0.0, 0.2, 1],        // standard ease out
    inOut: [0.76, 0, 0.24, 1],      // cinematic, used for page transitions

  // NUMBER COUNTER HOOK — for animating stat numbers
  export function useCountUp(target: number, duration: number = 1200, 
                              delay: number = 0) {
    const [value, setValue] = useState(0)
    useEffect(() => {
      let startTime: number | null = null
      let frameId: number
      const delayTimeout = setTimeout(() => {
        const step = (timestamp: number) => {
          if (!startTime) startTime = timestamp
          const progress = Math.min((timestamp - startTime) / duration, 1)
          const eased = 1 - Math.pow(1 - progress, 3) // ease out cubic
          setValue(Math.floor(eased * target))
          if (progress < 1) frameId = requestAnimationFrame(step)
          else setValue(target)
        }
        frameId = requestAnimationFrame(step)
      }, delay)
      return () => { clearTimeout(delayTimeout); cancelAnimationFrame(frameId) }
    }, [target, duration, delay])
    return value
  }

  // BAR FILL HOOK — returns 0 to width% for progress bars
  export function useBarFill(targetPercent: number, delay: number = 0) {
    const [width, setWidth] = useState(0)
    useEffect(() => {
      const timeout = setTimeout(() => setWidth(targetPercent), delay + 100)
      return () => clearTimeout(timeout)
    }, [targetPercent, delay])
    return width
  }

## STEP 1 — PAGE TRANSITION WRAPPER

Find the root layout component that wraps every page 
(likely App.tsx, Layout.tsx, or wherever your router renders children).

Wrap the page outlet/children with this exact pattern:

  <AnimatePresence mode="wait">
    <motion.div
      key={currentRoute}        // use the current pathname as key
      initial={{ opacity: 0, y: 12 }}
      animate={{ opacity: 1, y: 0 }}
      exit={{ opacity: 0, y: -8 }}
      transition={{ duration: 0.25, ease: [0.19, 1, 0.22, 1] }}
      style={{ width: '100%', height: '100%' }}
    >
      {children}
    </motion.div>
  </AnimatePresence>

The key={currentRoute} is critical — without it AnimatePresence won't 
detect page changes. Use useLocation() from react-router to get it.
Exit animation is subtle upward — not a full slide, just 8px up and fade.
Enter animation is 12px up from below and fade in. Duration: 250ms.
This should feel like a clean page flip, not a dramatic slide show.

## GLOBAL RULES FOR ALL ANIMATIONS

### Performance rules — follow these exactly:
- Every motion.div that animates position MUST have: style={{ willChange: 'transform' }}
  Add this only to elements that actually move (not fade-only elements)
- Never animate width/height directly — use scaleX/scaleY or CSS transitions 
  for bars and panels (exception: progress bars use CSS transition on width, 
  this is fine and intentional)
- Never use motion on elements that repeat in a list of more than 20 items
  (if a student list has 30+ rows, animate only the container fadeIn, 
  not each individual row)

### Accessibility:
Wrap all animation logic with a prefers-reduced-motion check.
In your animations.ts file add at the top:
  export const prefersReducedMotion = 
    typeof window !== 'undefined' 
    ? window.matchMedia('(prefers-reduced-motion: reduce)').matches 
    : false

Then in every component that uses framer-motion variants:
  const shouldAnimate = !prefersReducedMotion
  initial={shouldAnimate ? "hidden" : false}
  animate={shouldAnimate ? "show" : false}

### Timing rules — never deviate:
- No single animation duration exceeds 900ms
- No entrance delay exceeds 800ms from page load
- Stagger between sibling items: 60ms to 120ms (never more)
- Counter animations: 600ms to 1400ms depending on size of number
- Page transition in: exactly 250ms
- Page transition out: exactly 180ms
- Hover responses: 120ms to 200ms maximum
- Spring animations: stiffness 400 damping 30 (snap) or 200/25 (slug)

### What to NEVER animate:
- Do NOT animate the sidebar itself sliding in/out
- Do NOT animate the top header bar
- Do NOT animate text that is longer than one line (paragraph text)
- Do NOT add any looping/idle animations anywhere
- Do NOT add scroll-triggered animations (everything animates on mount)
- Do NOT animate color changes with framer-motion (use CSS transition for color)
- Do NOT add any animation to form inputs themselves (labels, placeholders)
- Do NOT use AnimatePresence on anything except page transitions and 
  conditional renders (like modals, dropdowns if they exist)
